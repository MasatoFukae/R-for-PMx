nm_stats <- nm_data %>% group_by(ID) %>% slice(1) %>%
  mutate_if(is.double, ~signif(., 3)) %>% 
  group_by(PN) %>% 
  summarise_at(c("WT", "AGE", "CRCL", "HT"),
              list(~n(), ~min(.), ~max(.), ~mean(.), ~sd(.), ~median(.))) %>% 
  gather(val=Value, key=Key, -PN) %>% 
  separate(Key, c("Covariate", "Stats"), "_") %>% 
  spread(key=PN, value=Value) %>% 
  mutate(Stats     = factor(Stats, levels = c("n", "mean", "sd", "median", "min", "max")),
         Covariate = factor(Covariate, levels = c("AGE", "WT", "HT", "CRCL"))) %>%
